#!/usr/bin/env python3
"""测试健康检查端点"""
import requests
import json
import sys

BASE = "http://localhost:58100"

def test_health():
    print("=== 测试 GET /health ===")
    resp = requests.get(f"{BASE}/health", timeout=5)
    data = resp.json()
    print(f"状态码: {resp.status_code}")
    print(json.dumps(data, indent=2, ensure_ascii=False))
    
    assert resp.status_code == 200, "健康检查应返回 200"

    print("✅ PASS")
    return True

if __name__ == "__main__":
    try:
        test_health()
    except Exception as e:
        print(f"❌ FAIL: {e}")
        sys.exit(1)

### 脚本3: test_chat.py — 对话流程完整测试

```python
#!/usr/bin/env python3
"""
CPQ Agent 对话流程完整测试
测试所有核心工具在真实对话中的调用
"""
import requests
import json
import sys
import time

BASE = "http://localhost:58100"

def sse_chat(message: str, timeout: int = 60):
    """发送消息并收集 SSE 流式响应"""
    print(f"\n{'='*60}")
    print(f"👤 用户: {message}")
    print(f"{'='*60}")
    
    resp = requests.post(
        f"{BASE}/agent/chat",
        json={"messages": [{"role": "user", "content": message}]},
        stream=True,
        timeout=timeout,
        headers={"Accept": "text/event-stream"}
    )
    
    if resp.status_code != 200:
        print(f"❌ HTTP {resp.status_code}: {resp.text}")
        return False
    
    full_text = []
    events = []
    
    for line in resp.iter_lines(decode_unicode=True):
        if not line:
            continue
        
        if line.startswith("data: "):
            try:
                data = json.loads(line[6:])
                event_type = data.get("type", "unknown")
                events.append(data)
                
                if event_type == "message_delta":
                    text = data.get("content", "")
                    print(text, end="", flush=True)
                    full_text.append(text)
                    
                elif event_type == "status":
                    status = data.get("status", "")
                    content = data.get("content", "")
                    print(f"\n[{status}] {content}")
                    
                elif event_type == "done":
                    print(f"\n\n[DONE] turns={data.get('turns')}, tokens={data.get('tokens')}")
                    break
                    
            except json.JSONDecodeError:
                pass
    
    print()
    return True, "".join(full_text), events


def test_product_search():
    """测试1: 产品搜索"""
    print("\n" + "="*60)
    print("测试1: 产品搜索 - 自然语言查询")
    print("="*60)
    ok, text, events = sse_chat("帮我找一下高压储能产品", timeout=60)
    assert ok, "对话应成功完成"
    assert len(text) > 20, f"响应太短: {len(text)} 字符"
    # 检查是否调用搜索工具
    tool_calls = [e for e in events if e.get("type") in ("tool_call", "tool_call_start")]
    print(f"  → 工具调用次数: {len(tool_calls)}")
    if tool_calls:
        for tc in tool_calls:
            print(f"    - {tc.get('tool_name', tc.get('name', 'unknown'))}")
    print("✅ 测试1 PASS")
    return True


def test_config_validation():
    """测试2: 配置验证"""
    print("\n" + "="*60)
    print("测试2: 配置验证 - 验证规则检查")
    print("="*60)
    ok, text, events = sse_chat("帮我验证一下 modelId=2091 的 HVI 配置是否合规", timeout=60)
    assert ok, "对话应成功完成"
    assert len(text) > 20
    print("✅ 测试2 PASS")
    return True


def test_bom_and_pricing():
    """测试3: BOM + 定价"""
    print("\n" + "="*60)
    print("测试3: BOM 展开 + 定价 - 完整定价流程")
    print("="*60)
    ok, text, events = sse_chat(
        "把 modelId=2091 的产品做个完整的 BOM 展开和定价",
        timeout=60
    )
    assert ok, "对话应成功完成"
    assert len(text) > 30, f"响应太短: {len(text)} 字符"
    print("✅ 测试3 PASS")
    return True


def test_full_workflow():
    """测试4: 完整配单流程（端到端）"""
    print("\n" + "="*60)
    print("测试4: 端到端配单 - 完整工作流")
    print("="*60)
    
    # 多轮对话
    messages = [
        "德国客户需要家用高压储能系统，40kWh左右，VDE认证",
        "确认选择这个配置，帮我出报价",
    ]
    
    conversation = []
    for i, msg in enumerate(messages):
        ok, text, events = sse_chat(msg, timeout=90)
        assert ok, f"第{i+1}轮对话应成功"
        conversation.append(text)
    
    print("\n✅ 测试4 PASS (端到端工作流)")
    return True


def test_reverse_matching():
    """测试5: 价格反向匹配"""
    print("\n" + "="*60)
    print("测试5: 反向匹配 - 价格驱动产品推荐")
    print("="*60)
    ok, text, events = sse_chat(
        "客户预算只有 25000，推荐一个满足德国市场的低压储能方案",
        timeout=60
    )
    assert ok
    assert len(text) > 20
    print("✅ 测试5 PASS")
    return True


if __name__ == "__main__":
    print("\n🚀 CPQ Agent App 后端完整测试套件")
    print(f"后端地址: {BASE}")
    print(f"开始时间: {time.strftime('%Y-%m-%d %H:%M:%S')}")
    
    results = []
    try:
        results.append(("1. 产品搜索", test_product_search()))
    except Exception as e:
        print(f"❌ 测试1 FAIL: {e}")
        results.append(("1. 产品搜索", False))
    
    try:
        results.append(("2. 配置验证", test_config_validation()))
    except Exception as e:
        print(f"❌ 测试2 FAIL: {e}")
        results.append(("2. 配置验证", False))
    
    try:
        results.append(("3. BOM+定价", test_bom_and_pricing()))
    except Exception as e:
        print(f"❌ 测试3 FAIL: {e}")
        results.append(("3. BOM+定价", False))
    
    try:
        results.append(("4. 端到端", test_full_workflow()))
    except Exception as e:
        print(f"❌ 测试4 FAIL: {e}")
        results.append(("4. 端到端", False))
    
    try:
        results.append(("5. 反向匹配", test_reverse_matching()))
    except Exception as e:
        print(f"❌ 测试5 FAIL: {e}")
        results.append(("5. 反向匹配", False))
    
    # 汇总
    print("\n" + "="*60)
    print("📊 测试汇总")
    print("="*60)
    passed = sum(1 for _, ok in results if ok)
    total = len(results)
    for name, ok in results:
        print(f"  {'✅' if ok else '❌'} {name}")
    print(f"\n通过: {passed}/{total}")
    print(f"结束时间: {time.strftime('%Y-%m-%d %H:%M:%S')}")
    
    sys.exit(0 if passed == total else 1)
```

## 三、手动测试检查项

| # | 测试项 | 验证点 | 方法 |
|---|--------|--------|------|
| 1 | 后端启动 | uvicorn 无报错，端口 58100 监听 | `curl localhost:58100/health` |
| 2 | 健康检查 | 返回 CPQ 连接状态 | `python3 tests/test_health.py` |
| 3 | 配置 CRUD | GET/PUT /config 正常 | `python3 tests/test_config.py` |
| 4 | CPQ 连接测试 | 能 ping 通 CPQ App | `curl localhost:30000/` |
| 5 | 模型连接测试 | 能调用 DeepSeek API | POST /config/test-model |
| 6 | SSE 流式响应 | 消息逐字返回 | `python3 tests/test_chat.py` |
| 7 | 工具调用 | 对话中触发 search_product 等工具 | 检查日志 |
| 8 | 错误处理 | 断网、CPQ 宕机时返回友好错误 | 手动停 CPQ 测试 |
| 9 | 并发 | 同时 3 个对话不冲突 | `&` 并行跑 test_chat |
