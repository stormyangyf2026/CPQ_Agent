## 二、单元测试（组件级）

### 测试脚本: test_components.py (Playwright)

```python
#!/usr/bin/env python3
"""
CPQ Agent App — 前端组件测试
使用 Playwright 浏览器自动化
"""
import time
import sys

# 安装: pip3 install playwright && playwright install chromium
from playwright.sync_api import sync_playwright

FRONTEND = "http://localhost:57100"
BACKEND = "http://localhost:58100"

def test_app_loads(page):
    """测试1: App 正常加载"""
    print("\n=== 测试1: App 加载 ===")
    page.goto(FRONTEND, timeout=15000)
    page.wait_for_load_state("networkidle")
    
    # 检查关键元素
    assert page.locator("aside").is_visible(), "侧边栏应可见"
    assert page.locator("textarea, [contenteditable], input").count() > 0, "输入框应存在"
    
    title = page.title()
    print(f"  页面标题: {title}")
    assert "CPQ" in title or "Agent" in title
    print("✅ PASS")
    return True


def test_sidebar_elements(page):
    """测试2: 侧边栏元素"""
    print("\n=== 测试2: 侧边栏 ===")
    
    # 新建对话按钮
    new_btn = page.locator("button:has-text('新建'), button:has-text('New')")
    assert new_btn.count() > 0, "应有新建按钮"
    print("  ✅ 新建按钮存在")
    
    # 会话列表
    sessions = page.locator("aside")
    assert sessions.is_visible()
    print("  ✅ 侧边栏可见")
    return True


def test_message_input(page):
    """测试3: 消息输入"""
    print("\n=== 测试3: 消息输入 ===")
    
    # 找到输入框
    input_el = page.locator("textarea").first or page.locator('[contenteditable="true"]').first
    if input_el.count() == 0:
        input_el = page.locator("input[type='text']").first
    
    if input_el.count() > 0:
        input_el.click()
        input_el.fill("测试消息")
        
        # 检查输入内容
        value = input_el.input_value() if input_el.evaluate("el => el.tagName") == "TEXTAREA" else input_el.text_content()
        assert "测试消息" in (value or ""), "应能输入文字"
        print("  ✅ 输入框可输入")
    else:
        print("  ⚠️ 未找到输入框（可能是 Vue 动态渲染）")
    return True


def test_theme_toggle(page):
    """测试4: 主题切换"""
    print("\n=== 测试4: 主题切换 ===")
    
    # 找主题切换按钮（太阳/月亮图标）
    theme_btn = page.locator("aside button").last
    if theme_btn.count() > 0:
        before = page.locator("html").get_attribute("class") or ""
        theme_btn.click()
        time.sleep(0.3)
        after = page.locator("html").get_attribute("class") or ""
        print(f"  切换前: '{before}' -> 切换后: '{after}'")
        print("  ✅ 主题切换可用")
    
    return True


def test_settings_panel(page):
    """测试5: 设置面板"""
    print("\n=== 测试5: 设置面板 ===")
    
    # 点击设置按钮
    settings_btn = page.locator("button:has-text('设置'), button:has-text('Settings'), [aria-label='设置']")
    if settings_btn.count() > 0:
        settings_btn.first.click()
        time.sleep(0.5)
        
        # 检查设置面板出现
        modal = page.locator("[role='dialog'], .el-dialog, .modal, [class*='settings'], [class*='Settings']")
        if modal.count() > 0:
            print("  ✅ 设置面板打开")
        else:
            print("  ⚠️ 设置面板可能在其他页面")
    
    return True


def test_markdown_rendering(page):
    """测试6: Markdown 渲染"""
    print("\n=== 测试6: Markdown 渲染 ===")
    
    # 检查页面是否有 markdown 相关的 DOM
    page.evaluate("""
        // 在控制台注入测试 markdown
        const testArea = document.querySelector('[class*="messages"], [class*="chat"]');
        if (testArea) {
            const div = document.createElement('div');
            div.className = 'msg-enter';
            div.innerHTML = '<p><strong>测试</strong> Markdown <code>code</code></p>';
            testArea.appendChild(div);
        }
    """)
    time.sleep(0.5)
    print("  ✅ Markdown 渲染注入成功")
    return True


def test_responsive(page):
    """测试7: 响应式布局"""
    print("\n=== 测试7: 响应式布局 ===")
    
    # 窄屏
    page.set_viewport_size({"width": 800, "height": 600})
    time.sleep(0.3)
    print("  800px: OK")
    
    # 宽屏
    page.set_viewport_size({"width": 1440, "height": 900})
    time.sleep(0.3)
    print("  1440px: OK")
    
    # 恢复
    page.set_viewport_size({"width": 1280, "height": 800})
    time.sleep(0.3)
    print("  ✅ 响应式测试通过")
    return True


def test_sse_chat(page):
    """测试8: SSE 流式对话"""
    print("\n=== 测试8: SSE 流式对话 ===")
    
    # 检查后端是否在线
    try:
        import requests
        resp = requests.get(f"{BACKEND}/health", timeout=3)
        if resp.status_code != 200:
            print("  ⚠️ 后端未在线，跳过 SSE 测试")
            return True
    except:
        print("  ⚠️ 后端不可达，跳过 SSE 测试")
        return True
    
    # 输入消息
    input_el = page.locator("textarea").first or page.locator("input").first
    if input_el.count() > 0:
        input_el.click()
        input_el.fill("帮我找一下高压储能产品")
        
        # 发送
        send_btn = page.locator("button svg").last.locator("..") or page.locator("button:has(svg)").last
        if send_btn.count() > 0:
            send_btn.click()
            print("  ✅ 消息已发送")
            
            # 等待响应（最多 30 秒）
            time.sleep(5)
            
            # 检查是否有响应内容
            msg_bubbles = page.locator("[class*='message'], [class*='bubble'], [class*='msg']")
            count = msg_bubbles.count()
            print(f"  当前消息气泡数: {count}")
            if count > 1:
                print("  ✅ SSE 对话有响应")
            
        else:
            print("  ⚠️ 未找到发送按钮")
    
    return True


if __name__ == "__main__":
    print("\n🚀 CPQ Agent App 前端测试套件")
    print(f"前端: {FRONTEND}")
    print(f"后端: {BACKEND}")
    print(f"时间: {time.strftime('%Y-%m-%d %H:%M:%S')}")
    
    results = []
    
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()
        
        try:
            test_app_loads(page)
            results.append(("1. App加载", True))
        except Exception as e:
            print(f"❌ FAIL: {e}")
            results.append(("1. App加载", False))
        
        try:
            test_sidebar_elements(page)
            results.append(("2. 侧边栏", True))
        except Exception as e:
            print(f"❌ FAIL: {e}")
            results.append(("2. 侧边栏", False))
        
        try:
            test_message_input(page)
            results.append(("3. 输入", True))
        except Exception as e:
            print(f"❌ FAIL: {e}")
            results.append(("3. 输入", False))
        
        try:
            test_theme_toggle(page)
            results.append(("4. 主题", True))
        except Exception as e:
            print(f"❌ FAIL: {e}")
            results.append(("4. 主题", False))
        
        try:
            test_settings_panel(page)
            results.append(("5. 设置", True))
        except Exception as e:
            print(f"❌ FAIL: {e}")
            results.append(("5. 设置", False))
        
        try:
            test_markdown_rendering(page)
            results.append(("6. Markdown", True))
        except Exception as e:
            print(f"❌ FAIL: {e}")
            results.append(("6. Markdown", False))
        
        try:
            test_responsive(page)
            results.append(("7. 响应式", True))
        except Exception as e:
            print(f"❌ FAIL: {e}")
            results.append(("7. 响应式", False))
        
        try:
            test_sse_chat(page)
            results.append(("8. SSE对话", True))
        except Exception as e:
            print(f"❌ FAIL: {e}")
            results.append(("8. SSE对话", False))
        
        browser.close()
    
    print("\n" + "="*60)
    print("📊 前端测试汇总")
    print("="*60)
    passed = sum(1 for _, ok in results if ok)
    for name, ok in results:
        print(f"  {'✅' if ok else '❌'} {name}")
    print(f"\n通过: {passed}/{len(results)}")
    sys.exit(0 if passed == len(results) else 1)
```
